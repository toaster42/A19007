018public class Stations {
019    // Declare ArrayList Objects
020    public ArrayList<String> staList = new ArrayList<>();
021    public ArrayList <String> staIndex = new ArrayList<>();
022    String strRecord;
023    String strFilePath;
024    int intCount;
025
026    /**
027     * The purpose of this constructor is to accept the winds aloft filename as an argument
028     * and set it as a class variable before opening the file.  Then, if the file exists it reads the data into
029     * two ArrayLists, staList for the data and staIndex for the Station IDs. Finally, it closes the input file.
030     *
031     * @param strInputFileName The Winds Aloft data file
032     * @throws IOException Does nothing
033     */
034    public Stations(String strInputFileName) throws IOException{
035        strFilePath=strInputFileName;
036
037        // Open file for input (pg 248)
038        File myFile = new File(strFilePath);
039        if (myFile.exists())
040        {
041            // Yea! File Exists (pg 249)
042            Scanner inputFile = new Scanner(myFile);
043            // Initialize a Counter
044            intCount = 0;
045            while (inputFile.hasNext())
046            {
047                strRecord = inputFile.nextLine();
048
049                if (intCount > 7) {
050                    //write station weather to ArrayList
051                    staList.add(strRecord);
052
053                    //write StationID to Index Arraylist
054                    staIndex.add(strRecord.substring(0, 3));
055
056                }
057                intCount++;
058            }
059            inputFile.close();
060        }
061    }
062
063    /**The purpose of this method is to return the StationID at ArrayList staIndex's given ListPosition.
064     *
065     * @param intPosition the ordinal within ArrayList staIndex
066     * @return strStationID - the 3 letter code for the airstation
067     */
068    public String getStationID(int intPosition) {
069        String strStationID = staIndex.get(intPosition);
070        return strStationID;
071    }
072    /**The purpose of this method is to return number of stations in ArrayList staIndex.
073     *
074     * @return staIndex.size()
075     */
076    public int length() {return staIndex.size();}
077
078    /**The purpose of this method is to return true if the station exists in ArrayList staIndex.
079     *
080     * @param strStationID The 3 character Station ID
081     * @return staIndex.contains(strStationID) True or False
082     */
083    public Boolean exists(String strStationID) { return staIndex.contains(strStationID); }
084
085    /**The purpose of this method is to return the full station weather string from ArrayList staList. If it
086     * does not exist it returns "Station Not Found."
087     *
088     * @param strStationID The 3 character Station ID
089     * @return strStationData The Station Weather data
090     */
091    public String getStaWea(String strStationID) {
092        String strStationData = null;
093        int intPos;
094        if (exists(strStationID)) {
095            intPos = staIndex.indexOf(strStationID);
096            strStationData = staList.get(intPos);
097        }
098        else {
099            strStationData = "Station Not Found";
100        }
101        return strStationData;
102
103    }
104}
