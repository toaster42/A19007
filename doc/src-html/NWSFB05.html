013public class NWSFB05 {
014    private String strWeather;
015    public NWSFB05(String strVar) {
016        strWeather = strVar + " ";
017    }
018
019
020    /**
021     * The purpose of this method is to return the
022     * first three characters of the station weather.
023     * @return strStatopmID - first three characters of station weather
024     */
025    public String getStationID() {
026        String strStationID = strWeather.substring(0, 3);
027        return strStationID;
028    }
029
030    /**The purpose of this method is to accept a string representing an altitude and return an integer representing
031     * the starting character position of the altitude weather in the station weather string.
032     * @param strAlt The specified altitude
033     * @return intPos - the character position in the Station Data String
034     */
035    public static int getPos(String strAlt) {
036        //utilize a switch structure to determine which altitude was sent. and where to set 'cursor' for string clipping
037        int intPos = 0;
038
039        switch (strAlt) {
040            case "03000":   intPos = 4;
041                break;
042            case "06000":   intPos = 9;
043                break;
044            case "09000":   intPos = 17;
045                break;
046            case "12000":  intPos = 25;
047                break;
048            case "18000":  intPos = 33;
049                break;
050            case "24000":  intPos = 41;
051                break;
052            case "30000":  intPos = 49;
053                break;
054            case "34000":  intPos = 56;
055                break;
056            case "39000":  intPos = 63;
057                break;
058            default:    intPos = 0;
059                break;
060        }
061
062        return intPos;
063    }
064
065    /**
066     * The purpose of this method is to call getPos(strAlt) to determine where to begin extracting
067     * weather data, then extracts the data from the string.
068     * @param strAlt The specified altitude
069     * @return strAltitudeWeather - The weather data for the requested altitude
070     */
071    public String getAltitudeWeather(String strAlt) {
072        String strAltitudeWeather = "";
073        int intPos=getPos(strAlt);
074
075        if (intPos==4) {
076            strAltitudeWeather = strWeather.substring(intPos, intPos+4) + "   ";
077        }
078        else if (intPos==0) {
079            System.out.println("Invalid Altitude. Valid values:");
080            System.out.println("3, 6, 9, 12, 18, 24, 30, 34, 39");
081        }
082        else {
083            strAltitudeWeather = strWeather.substring(intPos, intPos + 7);
084        }
085
086        return strAltitudeWeather;
087    }
088    /**
089     * The purpose of this method is to accept a two character string representing the altitude
090     * and return the wind direction as a string.<br><br>
091     * Exceptions:<br>
092     *      1. If the first two characters are blanks, return "N/A"<br>
093     *      2. If the first four characters of the altitude weather are "9900" then return "Calm"<br>
094     *      3. If the first two characters of the Altitude Weather are greater than 36 then subtract 50.
095     *          This happens when the wind is blowing over 100 knots and the NWS needed a condition to
096     *          indicate that.  For example, 7715-12  would indicate that the wind is coming from 270
097     *          degrees at 115 knots at -12C.  (77-50 = 27).  Because the two characters represent the
098     *          wind direction, and something like 27 actually means 270 degrees, there is no such thing
099     *          as 39 or 390 degrees.  Thus, the National Weather Service will use 01 through 36 and 51 through 86
100     *          only, they do not code it with numbers that start with anything between 37 and 49 inclusive.<br><br>
101     * Otherwise just return the first two characters with a "0" appended to it.
102     * @param strAlt The two digit altitude (in thousands of feet)
103     * @return strWindDirection - The formatted Wind Direction
104     */
105    public String getWindDirection(String strAlt) {
106        String strAltitudeWeather = getAltitudeWeather(strAlt);
107        //System.out.println("strAltitudeWeather = " + strAltitudeWeather); //Used in testing
108        String strWindDirection;
109        //System.out.println("strAltitudeWeather.substring(0,2) = " + strAltitudeWeather.substring(0,2)); //Used in testing
110        if (strAltitudeWeather.substring(0,2).equals("  ")) {
111            strWindDirection = "N/A        ";
112        }
113        else if (strAltitudeWeather.substring(0,4).equals("9900")) {
114            strWindDirection = "Calm       ";
115        }
116        else if (Integer.parseInt(strAltitudeWeather.substring(0,2)) > 36) {
117            int intWindDirection = Integer.parseInt(strAltitudeWeather.substring(0,2)) - 50;
118            strWindDirection = String.valueOf(intWindDirection) + "0 degrees";
119        }
120        else {
121            strWindDirection = strAltitudeWeather.substring(0,2) + "0 degrees";
122        }
123        return strWindDirection;
124    }
125    /**
126     * The purpose of this method is to accept a two character string representing the
127     * altitude in thousands and return the wind speed as a string.<br><br>
128     * Exceptions:<br>
129     *      If the third and fourth characters of the altitude weather are blanks, return "N/A"<br>
130     *      If the first four characters of the altitude weather are "9900" then return "Calm"<br>
131     *      If the first two characters of the Altitude Weather are greater then 36 then add 100 to returned speed.<br><br>
132     *      Otherwise just return the second two characters of the altitude weather.<br>
133     * @param strAlt The two digit altitude (in thousands of feet)
134     * @return strWindSpeed - The formatted Wind Speed
135     */
136    public String getWindSpeed(String strAlt) {
137        String strAltitudeWeather = getAltitudeWeather(strAlt);
138        String strWindSpeed;
139
140        if ((strAltitudeWeather.charAt(3) == ' ') || (strAltitudeWeather.charAt(4) == ' ')) {
141            strWindSpeed = "N/A      ";
142        }
143        else if (strAltitudeWeather.substring(0,4).equals("9900")) {
144            strWindSpeed = " Calm    ";
145        }
146
147        //Add 0 after speed
148        else if (Integer.parseInt(strAltitudeWeather.substring(0,2)) > 36) {
149            strWindSpeed = strAltitudeWeather.substring(2, 4) + "0 knots";
150        }
151        else {
152            strWindSpeed = " " + strAltitudeWeather.substring(2, 4) + " knots";
153        }
154        return strWindSpeed;
155    }
156
157    /**
158     * The purpose of this method is to accepts a two character string
159     * representing the altitude and returns the wind temperature as a
160     * string.<br><br>
161     * If the fifth character is a blank, it returns "N/A."<br>
162     * If the altitude is over 24000 feet, it adds a "-" as a prefix.<br><br>
163     * Otherwise it just returns the last three characters.
164     * @param strAlt The two digit altitude (in thousands of feet)
165     * @return strWindTemp - The formatted Altitude Temperature
166     */
167    public String getWindTemp(String strAlt) {
168        String strAltitudeWeather = getAltitudeWeather(strAlt);
169        String strWindTemp;
170
171        if (strAltitudeWeather.charAt(5) == ' ') {
172            strWindTemp = " N/A";
173        }
174        else if (Integer.parseInt(strAlt) > 24000) {
175            strWindTemp = "-" + strAltitudeWeather.substring(4, 6) + "C";
176        }
177        else {
178            strWindTemp = strAltitudeWeather.substring(4, 7) + "C";
179        }
180
181        return strWindTemp;
182    }
183
184    /**
185     * The purpose of this method is to return a formatted string of the Altitude Weather.<br><br>
186     * Example output:<br>Station ID: SAN<br>
187     The Altitude Weather for 3000 feet is: 3106<br>
188     Wind Direction: 310 degrees<br>
189     Wind Speed: 06 knots<br>
190     Wind Temperature: n/a<br>
191     * @param strAlt The two digit altitude (in thousands of feet)
192     * @return strFmtWeatherReport - The formatted Altitude Weather
193     */
194    public String fmtWeatherReport(String strAlt) {
195        return "Station ID: " + getStationID() + "\r\n" + "The Altitude Weather for " + strAlt +
196                ",000 feet is: " + getAltitudeWeather(strAlt) + "\r\n" + "Wind Direction: " + getWindDirection(strAlt)
197                + "\r\n" + "Wind Speed: " + getWindSpeed(strAlt) + "\r\n" + "Wind Temperature: " +
198                getWindTemp(strAlt);
199    }
200}
